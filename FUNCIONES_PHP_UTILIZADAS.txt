					################	FUNCIONES PHP UTILIZADAS O PROBADAS		#################


	####dirname($path,$levels) ::: ESTA FUNCION RECIBE UNA RUTA $path VALIDA Y DEVUELVE ESTA RUTA SUBIENDO TANTOS NIVELES SE HAYA INDICADO CON $levels ESTE PARAMETRO DEBE SER UN ENTERO MAYOR A 1(POR DEFECTO ES 1), POR EJEMPLO ... 
		* dirname("/usr/www/htdocs") DEVUELVE /usr/www 
		* dirname("/usr/www/htdocs",2) DEVUELVE /usr

	####__DIR__ ::: CONSTANTE PREDEFINIDA EN PHP. DEVUELVE LA RUTA EXACTA DEL DIRECTORIO PADRE DEL ARCHIVO QUE LA EJECUTA

	####is_string($string) ::: RECIBE UNA VARIABLE $string Y DEVUELVE VERDADERO SI ES DE TIPO CADENA DE TEXTO

	####settype($var,$type) ::: RECIBE UNA VARIABLE $var Y DEVUELVE VERDADERO CUANDO LA VARIABLE SE HA ESTABLECIDO AL TIPO INDICADO EN $type

	####json_encode($var) ::: DEVUELVE UNA REPRESENTACION EN FORMATO JSON DEL VALOR CONTENIDO EN LA VARIABLE $var
